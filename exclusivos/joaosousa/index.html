# =======================================================
# db_config.py - CONEXIÓN A MYSQL
# Este archivo se encarga ÚNICAMENTE de conectarse a la BD
# =======================================================

# 1. Importamos el conector de MySQL
#    (debe estar instalado con: pip install mysql-connector-python)
import mysql.connector

# 2. Función que crea y devuelve una conexión activa
#    Cada vez que la llamamos, obtenemos un objeto conexión listo para usar
def get_connection():
    # mysql.connector.connect() crea un objeto de tipo 'MySQLConnection'
    # que representa la sesión con el servidor MySQL.
    #
    # PARÁMETROS (cámbialos según tu configuración):
    # - host: "localhost" → MySQL está en la misma máquina
    # - user: "root" → tu usuario de MySQL
    # - password: "tu_contraseña" → tu contraseña de MySQL
    # - database: "crud_tkinter" → nombre de la BD (debes crearla antes)
    return mysql.connector.connect(
        host="localhost",
        user="root",
        password="tu_contraseña",    # ¡Cámbiala por tu contraseña real!
        database="crud_tkinter"
    )
    # Lo que devuelve: un objeto 'conn' que tiene métodos como:
    # .cursor() para crear cursores, .commit() para guardar, .close() para cerrar
🧠 Antes de continuar: Asegúrate de tener MySQL corriendo y haber creado la base de datos con CREATE DATABASE crud_tkinter; y la tabla usuarios con CREATE TABLE usuarios (id INT AUTO_INCREMENT PRIMARY KEY, nombre VARCHAR(100), email VARCHAR(100));

3. Archivo 2: crud_ops.py → Operaciones CRUD

📝 crud_ops.py CÓPIALO COMPLETO

📌 ¿Dónde lo pego? Crea un archivo llamado crud_ops.py dentro de la carpeta mi_crud_win y pega EXACTAMENTE este código: 

# =======================================================
# crud_ops.py - OPERACIONES CRUD (Create, Read, Update, Delete)
# Aquí están TODAS las funciones que interactúan con la BD
# =======================================================

# 1. Importamos la función de conexión desde db_config.py
from db_config import get_connection

# =======================================================
# 2. CREAR (INSERT) - Guarda un nuevo usuario
# =======================================================
def crear_usuario(nombre, email):
    # 2a. Obtenemos una conexión activa
    conn = get_connection()
    # 2b. Creamos un cursor para ejecutar SQL
    cursor = conn.cursor()
    # 2c. Definimos la consulta SQL (parametrizada para seguridad)
    sql = "INSERT INTO usuarios (nombre, email) VALUES (%s, %s)"
    # 2d. Ejecutamos la consulta con los datos
    cursor.execute(sql, (nombre, email))
    # 2e. Guardamos los cambios en la BD (commit)
    conn.commit()
    # 2f. Cerramos la conexión para liberar recursos
    conn.close()

# =======================================================
# 3. LEER (SELECT) - Obtiene todos los usuarios
# =======================================================
def obtener_usuarios():
    conn = get_connection()
    # dictionary=True → las filas vienen como diccionarios
    cursor = conn.cursor(dictionary=True)
    cursor.execute("SELECT * FROM usuarios")
    # fetchall() devuelve TODAS las filas
    resultados = cursor.fetchall()
    conn.close()
    return resultados

# =======================================================
# 4. ACTUALIZAR (UPDATE) - Modifica un usuario existente
# =======================================================
def actualizar_usuario(id, nombre, email):
    conn = get_connection()
    cursor = conn.cursor()
    sql = "UPDATE usuarios SET nombre=%s, email=%s WHERE id=%s"
    cursor.execute(sql, (nombre, email, id))
    conn.commit()
    conn.close()

# =======================================================
# 5. ELIMINAR (DELETE) - Borra un usuario por su ID
# =======================================================
def eliminar_usuario(id):
    conn = get_connection()
    cursor = conn.cursor()
    cursor.execute("DELETE FROM usuarios WHERE id=%s", (id,))
    conn.commit()
    conn.close()
🧠 ¿Qué hace este archivo? Contiene TODAS las funciones que necesitamos para manejar los datos. main.py llama a estas funciones cuando el usuario hace clic en los botones.

# =======================================================
# main.py - INTERFAZ GRÁFICA CON TKINTER
# Este archivo crea la ventana, los campos y los botones
# =======================================================

# 1. Importamos tkinter y sus módulos
import tkinter as tk
from tkinter import messagebox, ttk

# 2. Importamos las funciones CRUD desde crud_ops.py
from crud_ops import crear_usuario, obtener_usuarios, actualizar_usuario, eliminar_usuario

# =======================================================
# 3. FUNCIONES QUE RESPONDEN A LOS BOTONES
# =======================================================

# 3a. Guardar nuevo usuario (CREATE)
def guardar_datos():
    nombre = entry_nombre.get().strip()
    email = entry_email.get().strip()
    if nombre and email:
        crear_usuario(nombre, email)   # ← aquí viajan los datos a la BD
        messagebox.showinfo("✅ Éxito", "Usuario guardado correctamente")
        limpiar_campos()
        listar_usuarios()   # actualizamos la lista
    else:
        messagebox.showerror("❌ Error", "Completa todos los campos")

# 3b. Cargar usuario seleccionado en los campos (para editar)
def cargar_usuario(event):
    # Obtenemos el elemento seleccionado en el Treeview
    seleccion = tree.selection()
    if seleccion:
        item = tree.item(seleccion[0])
        valores = item['values']
        # Guardamos el ID en una variable global
        global id_seleccionado
        id_seleccionado = valores[0]
        entry_nombre.delete(0, tk.END)
        entry_nombre.insert(0, valores[1])
        entry_email.delete(0, tk.END)
        entry_email.insert(0, valores[2])
        # Cambiamos el botón Guardar por Actualizar
        btn_guardar.config(text="✏️ Actualizar", command=actualizar_datos)

# 3c. Actualizar usuario existente (UPDATE)
def actualizar_datos():
    nombre = entry_nombre.get().strip()
    email = entry_email.get().strip()
    if nombre and email and id_seleccionado:
        actualizar_usuario(id_seleccionado, nombre, email)
        messagebox.showinfo("✅ Éxito", "Usuario actualizado")
        limpiar_campos()
        listar_usuarios()
        btn_guardar.config(text="💾 Guardar", command=guardar_datos)

# 3d. Eliminar usuario (DELETE)
def eliminar_datos():
    if id_seleccionado:
        respuesta = messagebox.askyesno("⚠️ Confirmar", "¿Eliminar este usuario?")
        if respuesta:
            eliminar_usuario(id_seleccionado)
            messagebox.showinfo("✅ Éxito", "Usuario eliminado")
            limpiar_campos()
            listar_usuarios()
            btn_guardar.config(text="💾 Guardar", command=guardar_datos)

# 3e. Limpiar campos del formulario
def limpiar_campos():
    entry_nombre.delete(0, tk.END)
    entry_email.delete(0, tk.END)
    global id_seleccionado
    id_seleccionado = None
    btn_guardar.config(text="💾 Guardar", command=guardar_datos)

# 3f. Listar todos los usuarios en el Treeview
def listar_usuarios():
    # Limpiamos el Treeview
    for item in tree.get_children():
        tree.delete(item)
    # Obtenemos los datos de la BD
    usuarios = obtener_usuarios()
    for u in usuarios:
        tree.insert("", tk.END, values=(u['id'], u['nombre'], u['email']))

# =======================================================
# 4. CREACIÓN DE LA VENTANA Y WIDGETS
# =======================================================

# 4a. Ventana principal
ventana = tk.Tk()
ventana.title("📋 CRUD con Tkinter + MySQL")
ventana.geometry("700x550")
ventana.config(bg="#f0f4ff")

# 4b. Variable global para el ID seleccionado
id_seleccionado = None

# 4c. Frame para el formulario
frame_form = tk.Frame(ventana, bg="#e8f0fe", padx=20, pady=15, relief=tk.RIDGE, bd=2)
frame_form.pack(fill=tk.X, padx=20, pady=10)

# 4d. Labels y Entries
tk.Label(frame_form, text="👤 Nombre:", font=("Segoe UI", 11), bg="#e8f0fe").grid(row=0, column=0, sticky="w", pady=5)
entry_nombre = tk.Entry(frame_form, font=("Segoe UI", 11), width=30)
entry_nombre.grid(row=0, column=1, padx=10, pady=5)

tk.Label(frame_form, text="📧 Email:", font=("Segoe UI", 11), bg="#e8f0fe").grid(row=1, column=0, sticky="w", pady=5)
entry_email = tk.Entry(frame_form, font=("Segoe UI", 11), width=30)
entry_email.grid(row=1, column=1, padx=10, pady=5)

# 4e. Botones
frame_botones = tk.Frame(frame_form, bg="#e8f0fe")
frame_botones.grid(row=2, column=0, columnspan=2, pady=15)

btn_guardar = tk.Button(frame_botones, text="💾 Guardar", font=("Segoe UI", 10, "bold"),
                        bg="#2563eb", fg="white", padx=20, pady=8, command=guardar_datos)
btn_guardar.pack(side=tk.LEFT, padx=5)

btn_eliminar = tk.Button(frame_botones, text="🗑️ Eliminar", font=("Segoe UI", 10, "bold"),
                         bg="#dc2626", fg="white", padx=20, pady=8, command=eliminar_datos)
btn_eliminar.pack(side=tk.LEFT, padx=5)

btn_limpiar = tk.Button(frame_botones, text="🧹 Limpiar", font=("Segoe UI", 10, "bold"),
                        bg="#64748b", fg="white", padx=20, pady=8, command=limpiar_campos)
btn_limpiar.pack(side=tk.LEFT, padx=5)

# 4f. Treeview para mostrar la lista de usuarios
frame_lista = tk.Frame(ventana, bg="#f0f4ff")
frame_lista.pack(fill=tk.BOTH, expand=True, padx=20, pady=10)

tree = ttk.Treeview(frame_lista, columns=("ID", "Nombre", "Email"), show="headings", height=12)
tree.heading("ID", text="ID")
tree.heading("Nombre", text="Nombre")
tree.heading("Email", text="Email")
tree.column("ID", width=50, anchor="center")
tree.column("Nombre", width=200)
tree.column("Email", width=250)

# Scrollbar para el Treeview
scrollbar = ttk.Scrollbar(frame_lista, orient=tk.VERTICAL, command=tree.yview)
tree.configure(yscrollcommand=scrollbar.set)

tree.pack(side=tk.LEFT, fill=tk.BOTH, expand=True)
scrollbar.pack(side=tk.RIGHT, fill=tk.Y)

# 4g. Evento: al hacer clic en un usuario, cargarlo en los campos
tree.bind("<<TreeviewSelect>>", cargar_usuario)

# 4h. Cargar la lista al iniciar
listar_usuarios()

# 4i. Iniciar el bucle principal de Tkinter
ventana.mainloop()

🧭 El viaje de los datos (explicado)

1. Usuario escribe en Entry → entry_nombre.get() captura el texto.
2. Botón "Guardar" → ejecuta guardar_datos() (en main.py).
3. guardar_datos() → llama a crear_usuario(nombre, email) (desde crud_ops.py).
4. crud_ops.crear_usuario() → llama a get_connection() (desde db_config.py).
5. get_connection() → retorna un objeto conn (MySQLConnection).
6. Sobre conn, creamos cursor → cursor = conn.cursor().
7. Ejecutamos SQL → cursor.execute("INSERT ...", (nombre, email)).
8. conn.commit() → confirma la transacción en la BD.
9. conn.close() → liberamos la conexión.
10. Mensaje de éxito → el usuario ve confirmación y la lista se actualiza.

📊 Resumen: cómo se comunican los archivos

main.py ⬇️ crud_ops.py ⬇️ db_config.py ⬇️ 🐬 MySQL
🔄 El flujo es siempre: Interfaz → Lógica → Conexión → Base de Datos

✅ Buenas prácticas en Windows

  • Usa entorno virtual → evita conflictos de paquetes.
  • Separa responsabilidades → interfaz (main.py) / lógica (crud_ops.py) / conexión (db_config.py).
  • Maneja excepciones → try/except en operaciones de base de datos.
  • Cierra siempre la conexión → conn.close() para evitar fugas.
  • Usa parametrización → (%s) para evitar inyección SQL.
  • En Windows: si usas python y no funciona, prueba con py.
