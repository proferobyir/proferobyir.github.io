El Contrato Equivale a una Clase 
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract MaquinaCafe { // <-- EQUIVALE A: class MaquinaCafe
    
    // ATRIBUTOS DE CLASE (Variables de estado encapsuladas)
    address public owner;             // Propietario del contrato
    uint256 public cafesDisponibles; // Atributo numérico
    uint256 public precio;           // Atributo numérico

    // CONSTRUCTOR: Inicializa el estado al hacer el "new" (Deploy)
    constructor(uint256 _inicial) {
        owner = msg.sender; // Guarda quién instanció el objeto
        cafesDisponibles = _inicial;
        precio = 2;
    }

    // MÉTODO PÚBLICO (Modifica los atributos)
    function comprarCafe() public payable {
        // Validaciones de estado pre-ejecución
        require(msg.value >= precio, "No enviaste suficiente Ether");
        require(cafesDisponibles > 0, "Instancia sin existencias");
        
        cafesDisponibles -= 1; // Mutación del atributo
    }
}
Ciclo de Vida del Objeto en Red
ESTADO: CLASE NO INSTANCIADA (Sin desplegar)

Paso 1: Pasa argumentos al constructor e instancía la clase en la red.

Variables de Estado del Objeto:

• owner: null

• cafesDisponibles: 0

Paso 2: Invocar métodos del objeto ya desplegado (Enviar dinero de pago).

La clase estática espera ser compilada e instanciada en la Blockchain...
