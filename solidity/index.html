Contrato Inteligente en Solidity

Este código define las reglas matemáticas inmutables en la Blockchain de Ethereum. 

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract MaquinaCafe {
    // Estado del contrato (Memoria en la Blockchain)
    uint256 public precioCafe = 2; 
    uint256 public cafesDisponibles = 10;

    // Función que ejecuta la acción automáticamente
    function comprarCafe(uint256 _pago) public {
        require(_pago >= precioCafe, "Saldo insuficiente");
        require(cafesDisponibles > 0, "No queda cafe");
        
        cafesDisponibles -= 1;
        // ¡Acción completada de forma autónoma!
    }
}
