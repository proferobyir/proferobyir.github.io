Guía de Base: Sistema CRUD de Universidad

Frontend HTML/JS Vanilla, Backend Node.js Express y Base de Datos MySQL

Nota pedagógica: Este código está diseñado con fines educativos utilizando JavaScript estructurado básico y manipulación directa del DOM (Vanilla) para asegurar el entendimiento del flujo de datos sin abstracciones de frameworks.

1. Base de Datos (MySQL)

Script inicial para la creación de la base de datos y la estructura de la tabla estudiantes.

database.sql 
CREATE DATABASE sistema_universidad;
USE sistema_universidad;

CREATE TABLE estudiantes (
    cedula VARCHAR(20) PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    apellido VARCHAR(100) NOT NULL,
    seccion VARCHAR(10) NOT NULL,
    carrera VARCHAR(100) NOT NULL,
    fecha_ingreso DATE NOT NULL
);

3. Servidor Backend (Node.js + Express)

Configuración de la API REST con soporte de CORS y parsing de JSON incorporado.

server.js 
import express from 'express';
import mysql from 'mysql2/promise';
import cors from 'cors';

const app = express();
app.use(cors());
app.use(express.json());

// Configuración de la conexión a MySQL
const db = await mysql.createConnection({
    host: 'localhost',
    user: 'root',      // Cambia según tu configuración
    password: '',      // Cambia según tu configuración
    database: 'sistema_universidad'
});

console.log('Conectado exitosamente a MySQL');

// 1. OBTENER TODOS LOS ESTUDIANTES O FILTRAR
app.get('/api/estudiantes', async (req, res) => {
    const { seccion, cedula, nombre } = req.query;
    let query = 'SELECT * FROM estudiantes WHERE 1=1';
    const params = [];

    if (seccion) {
        query += ' AND seccion = ?';
        params.push(seccion);
    }
    if (cedula) {
        query += ' AND cedula = ?';
        params.push(cedula);
    }
    if (nombre) {
        query += ' AND nombre LIKE ?';
        params.push(`%${nombre}%`);
    }

    try {
        const [rows] = await db.query(query, params);
        res.json(rows);
    } catch (error) {
        res.status(500).json({ error: error.message });
    }
});

// 2. CREAR UN NUEVO ESTUDIANTE
app.post('/api/estudiantes', async (req, res) => {
    const { cedula, nombre, apellido, seccion, carrera, fecha_ingreso } = req.body;
    const query = 'INSERT INTO estudiantes (cedula, nombre, apellido, seccion, carrera, fecha_ingreso) VALUES (?, ?, ?, ?, ?, ?)';
    
    try {
        await db.query(query, [cedula, nombre, apellido, seccion, carrera, fecha_ingreso]);
        res.status(201).json({ message: 'Estudiante registrado con éxito' });
    } catch (error) {
        res.status(500).json({ error: error.message });
    }
});

// 3. ACTUALIZAR UN ESTUDIANTE
app.put('/api/estudiantes/:cedula', async (req, res) => {
    const { cedula } = req.params;
    const { nombre, apellido, seccion, carrera, fecha_ingreso } = req.body;
    const query = 'UPDATE estudiantes SET nombre = ?, apellido = ?, seccion = ?, carrera = ?, fecha_ingreso = ? WHERE cedula = ?';

    try {
        await db.query(query, [nombre, apellido, seccion, carrera, fecha_ingreso, cedula]);
        res.json({ message: 'Estudiante actualizado con éxito' });
    } catch (error) {
        res.status(500).json({ error: error.message });
    }
});

// 4. ELIMINAR UN ESTUDIANTE
app.delete('/api/estudiantes/:cedula', async (req, res) => {
    const { cedula } = req.params;
    const query = 'DELETE FROM estudiantes WHERE cedula = ?';

    try {
        await db.query(query, [cedula]);
        res.json({ message: 'Estudiante eliminado con éxito' });
    } catch (error) {
        res.status(500).json({ error: error.message });
    }
});

const PORT = 3000;
app.listen(PORT, () => {
    console.log(`Servidor corriendo en http://localhost:${PORT}`);
});

5. Control de Lógica Frontend (JavaScript Vanilla)

Consumo de la API con Async/Await e inyección interactiva de elementos en el DOM.

app.js 
const API_URL = 'http://localhost:3000/api/estudiantes';

let editMode = false;

document.getElementById('student-form').addEventListener('submit', guardarEstudiante);
document.addEventListener('DOMContentLoaded', () => cargarEstudiantes());

async function cargarEstudiantes() {
    const cedula = document.getElementById('search-cedula').value;
    const nombre = document.getElementById('search-nombre').value;
    const seccion = document.getElementById('search-seccion').value;

    let url = new URL(API_URL);
    if (cedula) url.searchParams.append('cedula', cedula);
    if (nombre) url.searchParams.append('nombre', nombre);
    if (seccion) url.searchParams.append('seccion', seccion);

    try {
        const response = await fetch(url);
        const estudiantes = await response.json();
        
        const tbody = document.getElementById('tabla-estudiantes');
        tbody.innerHTML = '';

        estudiantes.forEach(est => {
            const fecha = est.fecha_ingreso.split('T')[0];
            
            const tr = document.createElement('tr');
            tr.innerHTML = `
                \${est.cedula}
                \${est.nombre}
                \${est.apellido}
                \${est.seccion}
                \${est.carrera}
                \${fecha}
                
                    
                    
                
            `;
            tbody.appendChild(tr);
        });
    } catch (error) {
        console.error('Error al cargar estudiantes:', error);
    }
}

async function guardarEstudiante(e) {
    e.preventDefault();

    const estudianteData = {
        cedula: document.getElementById('cedula').value,
        nombre: document.getElementById('nombre').value,
        apellido: document.getElementById('apellido').value,
        seccion: document.getElementById('seccion').value,
        carrera: document.getElementById('carrera').value,
        fecha_ingreso: document.getElementById('fecha_ingreso').value,
    };

    let url = API_URL;
    let method = 'POST';

    if (editMode) {
        url = `\${API_URL}/\${estudianteData.cedula}`;
        method = 'PUT';
    }

    try {
        const response = await fetch(url, {
            method: method,
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(estudianteData)
        });

        if (response.ok) {
            alert(editMode ? 'Estudiante actualizado' : 'Estudiante registrado');
            restablecerFormulario();
            cargarEstudiantes();
        } else {
            const err = await response.json();
            alert('Error: ' + err.error);
        }
    } catch (error) {
        console.error('Error al guardar:', error);
    }
}

function prepararEdicion(cedula, nombre, apellido, seccion, carrera, fecha) {
    document.getElementById('cedula').value = cedula;
    document.getElementById('cedula').disabled = true;
    document.getElementById('nombre').value = nombre;
    document.getElementById('apellido').value = apellido;
    document.getElementById('seccion').value = seccion;
    document.getElementById('carrera').value = carrera;
    document.getElementById('fecha_ingreso').value = fecha;

    document.getElementById('btn-guardar').innerText = 'Actualizar Estudiante';
    document.getElementById('btn-guardar').style.backgroundColor = '#ffc107';
    document.getElementById('btn-guardar').style.color = 'black';
    editMode = true;
}

async function eliminarEstudiante(cedula) {
    if (!confirm(`¿Eliminar estudiante \${cedula}?`)) return;

    try {
        const response = await fetch(`\${API_URL}/\${cedula}`, { method: 'DELETE' });
        if (response.ok) {
            alert('Estudiante eliminado');
            cargarEstudiantes();
        }
    } catch (error) {
        console.error('Error al eliminar:', error);
    }
}

function restablecerFormulario() {
    document.getElementById('student-form').reset();
    document.getElementById('cedula').disabled = false;
    document.getElementById('btn-guardar').innerText = 'Guardar Estudiante';
    document.getElementById('btn-guardar').style.backgroundColor = '#28a745';
    document.getElementById('btn-guardar').style.color = 'white';
    editMode = false;
}

function limpiarFiltros() {
    document.getElementById('search-cedula').value = '';
    document.getElementById('search-nombre').value = '';
    document.getElementById('search-seccion').value = '';
    cargarEstudiantes();
}
