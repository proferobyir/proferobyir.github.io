-- =========================================================================
-- SCRIPT DE CREACIÓN Y CARGA DE BASE DE DATOS TIPO NORTHWIND
-- =========================================================================

CREATE DATABASE IF NOT EXISTS tienda_northwind
  CHARACTER SET utf8mb4 COLLATE utf8mb4_spanish_ci;
USE tienda_northwind;

SET FOREIGN_KEY_CHECKS = 0;
DROP TABLE IF EXISTS detalles_pedidos;
DROP TABLE IF EXISTS pedidos;
DROP TABLE IF EXISTS productos;
DROP TABLE IF EXISTS proveedores;
DROP TABLE IF EXISTS categorias;
DROP TABLE IF EXISTS clientes;
DROP TABLE IF EXISTS empleados;
DROP TABLE IF EXISTS transportistas;
SET FOREIGN_KEY_CHECKS = 1;

-- =========================================================================
-- 1. CREACIÓN DE TABLAS
-- =========================================================================

CREATE TABLE categorias (
    categoria_id INT AUTO_INCREMENT PRIMARY KEY,
    nombre_categoria VARCHAR(50) NOT NULL,
    descripcion TEXT
) ENGINE=InnoDB;

CREATE TABLE proveedores (
    proveedor_id INT AUTO_INCREMENT PRIMARY KEY,
    nombre_empresa VARCHAR(100) NOT NULL,
    nombre_contacto VARCHAR(50),
    puesto_contacto VARCHAR(50),
    direccion VARCHAR(100),
    ciudad VARCHAR(50),
    region VARCHAR(50),
    codigo_postal VARCHAR(20),
    pais VARCHAR(50),
    telefono VARCHAR(20)
) ENGINE=InnoDB;

CREATE TABLE productos (
    producto_id INT AUTO_INCREMENT PRIMARY KEY,
    nombre_producto VARCHAR(100) NOT NULL,
    proveedor_id INT,
    categoria_id INT,
    precio_unitario DECIMAL(10,2) DEFAULT 0.00,
    unidades_stock INT DEFAULT 0,
    FOREIGN KEY (proveedor_id) REFERENCES proveedores(proveedor_id),
    FOREIGN KEY (categoria_id) REFERENCES categorias(categoria_id)
) ENGINE=InnoDB;

CREATE TABLE clientes (
    cliente_id VARCHAR(5) PRIMARY KEY,
    nombre_empresa VARCHAR(100) NOT NULL,
    nombre_contacto VARCHAR(50),
    puesto_contacto VARCHAR(50),
    direccion VARCHAR(100),
    ciudad VARCHAR(50),
    region VARCHAR(50),
    codigo_postal VARCHAR(20),
    pais VARCHAR(50)
) ENGINE=InnoDB;

CREATE TABLE empleados (
    empleado_id INT AUTO_INCREMENT PRIMARY KEY,
    apellido VARCHAR(50) NOT NULL,
    nombre VARCHAR(50) NOT NULL,
    puesto VARCHAR(50),
    tratamiento VARCHAR(10),
    fecha_nacimiento DATE,
    fecha_contratacion DATE,
    direccion VARCHAR(100),
    ciudad VARCHAR(50),
    region VARCHAR(50),
    codigo_postal VARCHAR(20),
    pais VARCHAR(50),
    reporta_a INT,
    FOREIGN KEY (reporta_a) REFERENCES empleados(empleado_id)
) ENGINE=InnoDB;

CREATE TABLE transportistas (
    transportista_id INT AUTO_INCREMENT PRIMARY KEY,
    nombre_empresa VARCHAR(100) NOT NULL,
    telefono VARCHAR(20)
) ENGINE=InnoDB;

CREATE TABLE pedidos (
    pedido_id INT AUTO_INCREMENT PRIMARY KEY,
    cliente_id VARCHAR(5),
    empleado_id INT,
    fecha_pedido DATE,
    fecha_entrega DATE,
    transportista_id INT,
    flete DECIMAL(10,2) DEFAULT 0.00,
    nombre_destinatario VARCHAR(100),
    direccion_destinatario VARCHAR(100),
    ciudad_destinatario VARCHAR(50),
    pais_destinatario VARCHAR(50),
    FOREIGN KEY (cliente_id) REFERENCES clientes(cliente_id),
    FOREIGN KEY (empleado_id) REFERENCES empleados(empleado_id),
    FOREIGN KEY (transportista_id) REFERENCES transportistas(transportista_id)
) ENGINE=InnoDB;

CREATE TABLE detalles_pedidos (
    pedido_id INT,
    producto_id INT,
    precio_unitario DECIMAL(10,2) NOT NULL,
    cantidad INT NOT NULL DEFAULT 1,
    descuento DECIMAL(4,2) DEFAULT 0.00,
    PRIMARY KEY (pedido_id, producto_id),
    FOREIGN KEY (pedido_id) REFERENCES pedidos(pedido_id),
    FOREIGN KEY (producto_id) REFERENCES productos(producto_id)
) ENGINE=InnoDB;
